SET(TEST_NAMES
    drivers_test
)

MACRO(ADD_UNIT_TEST NAME)
    ADD_EXECUTABLE(io_${NAME} ${NAME})
    TARGET_LINK_LIBRARIES(io_${NAME}
        fern_io
        ${Boost_UNIT_TEST_FRAMEWORK_LIBRARY}
    )
    ADD_TEST(NAME io_${NAME} COMMAND io_${NAME})
ENDMACRO()

FOREACH(NAME ${TEST_NAMES})
    ADD_UNIT_TEST(${NAME})
ENDFOREACH()
