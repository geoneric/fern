SET(TEST_NAMES
    stack_test
    string_test
)

MACRO(ADD_UNIT_TEST NAME)
    ADD_EXECUTABLE(${NAME} ${NAME})
    TARGET_LINK_LIBRARIES(${NAME}
        ranally_core
        ${Boost_UNIT_TEST_FRAMEWORK_LIBRARY}
    )
    ADD_TEST(NAME ${NAME} COMMAND ${NAME})
ENDMACRO()

FOREACH(NAME ${TEST_NAMES})
    ADD_UNIT_TEST(${NAME})
ENDFOREACH()
